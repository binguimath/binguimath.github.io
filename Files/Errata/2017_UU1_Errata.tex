\title{Errata to \textit{Unitarity of The Modular Tensor Categories Associated to Unitary Vertex Operator Algebras, I}}
\author{Bin Gui}


\begin{document}\sloppy

\maketitle


\begin{enumerate}
\item In the published version, the annihilation operator $\mathcal Y_{\overline ii}$ is misprinted as $\mc Y_{\ovl{ii}}$. (See for instance Eq. (1.41) (1.42).) The notations in the preprints are correct.

\item The proof of Lem. 3.8-(a) is too frivolous. A more serious argument is as follows. One should first understand in which sense (3.20) holds, i.e.
\begin{align*}
\mc Y_\alpha(w^{(i)},f)w^{(j)}=\sum_{s\in\Zbb_V}\wht f(s)\mc Y_\alpha(w^{(i)},s)w^{(j)}
\end{align*}
At the beginning, we only know that it holds when evaluated with any vector of $W_k$. Therefore, it holds in the algebraic completion $\ovl{W_k}$ where the RHS converges in the $\sigma(\ovl{W_k},W_k)$-topology to the LHS.

Equivalently, if for each $\lambda\geq0$ we let $P_\lambda$ be the projection of $\ovl{W_k}$ onto its $L_0$-weight $\lambda$ subspace, then 
\begin{align*}
P_\lambda\mc Y_\alpha(w^{(i)},f)w^{(j)}=\sum_{s\in\Zbb_V}\wht f(s)P_\lambda\mc Y_\alpha(w^{(i)},s)w^{(j)}
\end{align*}
where the RHS is a finite sum. One can also replace $P_\lambda$ with $P_{\leq \lambda}$ where $P_{\leq \lambda}=\sum_{0\leq \mu\leq \lambda}P_{\mu}$, noting that all but finitely many $\mu$ in $[0,\lambda]$ satisfy that $P_\mu\neq0$. Namely
\begin{align}\label{eq1}
P_{\leq \lambda}\mc Y_\alpha(w^{(i)},f)w^{(j)}=\sum_{s\in\Zbb_V}\wht f(s)P_{\leq \lambda}\mc Y_\alpha(w^{(i)},s)w^{(j)}
\end{align}
Now one proves Lem. 3.8-(a) as follows. Similar to (3.21), one shows that
\begin{align}
\sum_{s\in\Zbb_V}\Vert \wht f(s)P_{\leq \lambda}\mc Y_\alpha(w^{(i)},s)w^{(j)}\Vert_p\leq M_p|f|_{V,|p|+t}\Vert w^{(j)}\Vert_{p+r}
\end{align}
for all $\lambda$, noting that the LHS is a finite sum. This, together with \eqref{eq1}, shows that
\begin{align*}
\Vert P_{\leq \lambda}\mc Y_\alpha(w^{(i)},f)w^{(j)}\Vert_p\leq M_p|f|_{V,|p|+t}\Vert w^{(j)}\Vert_{p+r}
\end{align*}
Since $\lambda$ is arbitrary, we conclude that $\mc Y_\alpha(w^{(i)},f)w^{(j)}$, a priori only a vector of $\ovl{W_k}$, belongs to $\mc H_k^p$ (for all $p$).
\end{enumerate}
